    sendAlert('ORANGE', `CRITICAL: ${stats.crop} acceptance ${stats.acceptanceRate * 100}% for 48+ hours`);
    recommendPriceAdjustment(offerId, 'increase by 10-15%');
  }
  
  // Red Alert: <40% acceptance OR 3+ crops <60%
  const lowAcceptanceCrops = getAllOffers().filter(o => o.acceptanceRate < 0.60);
  if (stats.acceptanceRate < 0.40 || lowAcceptanceCrops.length >= 3) {
    sendAlert('RED', `EMERGENCY: Consider rollback to cooperative model`);
    scheduleEmergencyMeeting();
  }
  
  // Nuclear: <30% acceptance → immediate rollback
  if (stats.acceptanceRate < 0.30) {
    sendAlert('NUCLEAR', `Immediate rollback required for ${stats.crop}`);
    disableMarketplacePricing(stats.crop);
  }
}

// Check every 6 hours
setInterval(checkAllOffers, 6 * 60 * 60 * 1000);
